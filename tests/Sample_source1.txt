procedure BMI {
    read height1;
    read height2;
    read height3;
    read weight1;
    read weight2;
    read weight3;

    averageHeight = (height1 + height2 + height3) / 3;
    averageWeight = (weight1 + weight2 + weight3) / 3;

    heightInMeter = averageHeight / 100;

    index = averageWeight / (heightInMeter * heightInMeter);
    print index;
}
